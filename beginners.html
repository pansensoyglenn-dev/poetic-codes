3. Basic Math

a = 10
b = 3

print(a + b)    # 13
print(a - b)    # 7
print(a * b)    # 30
print(a / b)    # 3.333...
print(a // b)   # 3   (integer division)
print(a % b)    # 1   (remainder)
print(a ** 2)   # 100 (power)

6. Lists (like arrays)

fruits = ["apple", "banana", "mango", "kiwi"]

print(fruits[0])      # apple
print(fruits[-1])     # kiwi (last item)

fruits.append("orange")
fruits.pop(1)         # removes banana

for fruit in fruits:
    print(fruit)

7. For & While Loops

# For loop (most common)
for i in range(5):
    print(i)          # 0 1 2 3 4

# While loop
count = 0
while count < 3:
    print("Hello")
    count += 1

8. Simple Function

def greet(name, age=20):
    return f"Hi {name}! You are {age} years old."

print(greet("Alex"))
print(greet("Mia", 17))
